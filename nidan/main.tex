% タイトル
\newcommand{\Title}[5]{
  \twocolumn[
  \begin{center}
    \vspace{3cm}

    \Huge{#1} \\ \vspace*{0.1cm} % タイトル
    \huge{#2} \\ \vspace*{0.4cm} % サブタイトル
    \Large{#3 （学籍番号：#4）} \\ \vspace*{0.1cm} % 氏名・学籍番号
    \Large{#5} \\ % 所属
  \end{center}
  \ \\
  ]
}
% 1cm字下げ
\newcommand{\hhh}{\hspace*{1cm}}
% 画像1枚 \imgi{width[cm]}{image path}{caption}{label}{position[h,t,b,p,etc]}

  \begin{figure}[#5]
    \centering
    \includegraphics[width=#1cm]{#2}
    \caption{#3}
    \label{#4}
  \end{figure}
}
